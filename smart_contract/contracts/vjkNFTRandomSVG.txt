pragma solidity >=0.4.16 <0.9.0;

import "@openzeppelin/contracts/token/ERC721/ERC721.sol";
import "@openzeppelin/contracts/token/ERC721/extensions/ERC721URIStorage.sol";
import "base64-sol/base64.sol";

contract vjkNFT is ERC721URIStorage {

    uint256 public tokenCounter;
    address public sender;

    uint256 public maxNumberOfPaths;
    uint256 public maxNumberOfPathCommands;
    uint256 public size;
    string[] public pathCommands;
    string[] public colors;

    event Transfer (address sender, string imageURI, uint256 timestamp);

    struct TransferStruct {
    address sender;
    string imageURI;
    uint256 timestamp;
    }

    constructor() ERC721("vjkNFT", "vjkNFT") {

        tokenCounter = 0;

        maxNumberOfPaths = 24;
        maxNumberOfPathCommands = 5;
        size = 480;
        pathCommands = ["M", "L"];
        colors = ["red", "blue", "green", "yellow", "black", "orange"];
    }

    TransferStruct[] transactions;

    function create() public {

        uint256 tokenId = tokenCounter;
        tokenCounter += 1;

        sender = msg.sender;
        _safeMint(sender, tokenId);

        string memory svg = generateSVG(); 
        string memory imageURI = svgToImageURI(svg);
        string memory tokenURI = formatTokenURI(imageURI);

        _setTokenURI(tokenId, tokenURI);

        transactions.push(TransferStruct(sender, imageURI, block.timestamp));
        emit Transfer(sender, imageURI, block.timestamp);  
    }

    //----------------- generateSVG Start -------------------
    function generateSVG() public pure returns (string memory finalSvg) {
        finalSvg = "<svg xmlns='http://www.w3.org/2000/svg' height='480' width='480'><path d='M150 0 L75 200 L225 200 Z' /></svg>";
    }
    //----------------- Tools --------------------
    function svgToImageURI(string memory svg) public pure returns (string memory) {
        string memory baseURL = "data:image/svg+xml;base64,";
        string memory svgBase64Encoded = Base64.encode(bytes(string(abi.encodePacked(svg))));
        return string(abi.encodePacked(baseURL,svgBase64Encoded));
    }
        function formatTokenURI(string memory imageURI) public pure returns (string memory) {

        return string(
                abi.encodePacked(
                    "data:application/json;base64,",
                    Base64.encode(
                        bytes(
                            abi.encodePacked(
                                '{"image":"',imageURI,'"}'
                            )
                        )
                    )
                )
            );
    }
    function uint2str(uint _i) internal pure returns (string memory _uintAsString) {
        if (_i == 0) {
            return "0";
        }
        uint j = _i;
        uint len;
        while (j != 0) {
            len++;
            j /= 10;
        }
        bytes memory bstr = new bytes(len);
        uint k = len;
        while (_i != 0) {
            k = k-1;
            uint8 temp = (48 + uint8(_i - _i / 10 * 10));
            bytes1 b1 = bytes1(temp);
            bstr[k] = b1;
            _i /= 10;
        }
        return string(bstr);
    }
    function gettokenCounter() public view returns (uint256) {
        return tokenCounter;
    }
    function getAllTransactions() public view returns (TransferStruct[] memory) {
        return transactions;
    }
}
